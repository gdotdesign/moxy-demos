This page shows the current state of the battery.

When the battery goes ciritcal, notifications are fired. (vibrate, beeps, alert) 

updateBattery= (event) ->
  $("bat").set 'text', event.level+"%"
  $("charge").set 'text', if event.isPlugged then 'Charging' else 'Not Charging'
window.addEventListener 'batterycritical', (event) ->
  updateBattery(event)
  unless event.isPlugged
    navigator.notification.vibrate 1000
    navigator.notification.beep 3
    navigator.notification.alert 'Battery Critical'
window.addEventListener 'batterylow', updateBattery
window.addEventListener 'batterystatus', updateBattery
